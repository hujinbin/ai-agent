快速开始

1. 通过CDN引入插件

<script src="https://cdn.example.com/ai-agent.min.js"></script>

2. 初始化插件

// 创建 AI Agent 实例
const aiAgent = new AIAgent({
  apiUrl: '/api/ai/chat',
  theme: 'light',
  position: 'bottom-right'
});
